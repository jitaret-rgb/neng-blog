{{- /* Shortcode: img Usage: {{< img src="image.jpg" alt="description" width="600" >}} Features: แปลงรูปภาพเป็น WebP อัตโนมัติ พร้อม srcset หลายขนาด */ -}} {{- $src := .Get "src" -}} {{- $alt := .Get "alt" | default "" -}} {{- $width := .Get "width" | default 800 -}} {{- $class := .Get "class" -}} {{- $loading := .Get "loading" | default "lazy" -}} {{- $resource := .Page.Resources.Get $src -}} {{- if not $resource -}} {{- $resource = resources.Get $src -}} {{- end -}} {{- if and $resource (reflect.IsImageResourceProcessable $resource) -}} {{/* สร้างรูปหลายขนาดในรูปแบบ WebP */}} {{- $sizes := slice 400 800 1200 1600 -}} {{- $srcset := slice -}} {{- range $sizes -}} {{- if le . $resource.Width -}} {{- $resized := $resource.Resize (printf "%dx webp" .) -}} {{- $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink .) -}} {{- end -}} {{- end -}} {{- $defaultImage := $resource.Resize (printf "%dx webp" $width) -}} {{- $originalImage := $resource -}} {{- else -}} {{- end -}}
